PREFIX ex: <http://dbpedia.org/ontology/>
SELECT DISTINCT ?y
 WHERE
 {
   {
   ?x a ex:Politician.
   ?x ?y ?z
   }
 UNION
  {?class rdf:type ex:Politician}
 }

LIMIT 100

2. 
PREFIX ex: <http://dbpedia.org/ontology/>
SELECT DISTINCT ?y
 WHERE
 {
   {
   ?x a ex:Politician.
   ?x ?y ?z
   }
 UNION
  {?class rdf:type ex:Politician}
FILTER (?y != rdf:type)
 }

LIMIT 100
